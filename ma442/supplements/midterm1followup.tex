\begin{question}
  \begin{itemize}
    \item[(a)] Produce a linear transformation $\sfT \colon \RR^2 \to \RR^2$ with the property that there does \textbf{not} exist a basis
  $\beta$ of $\RR^2$ such that $[\sfT]_\beta = \begin{bmatrix} a & 0 \\ 0 & 0 \end{bmatrix}$ for some scalar $a \in
  \RR$. (Hint: consider the transformation $\sfT$ defined by $\sfT(x,y) = (y,0)$.)
    \item[(b)] If $\sfT \colon \sfV \to \sfV$ is a linear transformation with $\dim \sfV = 2$ and $\dim \ker \sfT = 1$,
      then show that there exists bases $\beta,\gamma$ of $\sfV$ with the property that 
      \begin{equation}\label{}
        [\sfT]_\beta^\gamma = \begin{bmatrix} 1 & 0 \\ 0 & 0 \end{bmatrix} .
      \end{equation}
\end{itemize}
\end{question}
